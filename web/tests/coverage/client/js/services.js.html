SmartApp.service('PointsService', ['$http','$q', function($http, $q){
	return {
		'getPoints': function(page) {
			var defer=$q.defer();
			$http.get('/points/getPoints?page='+(page+1)).success(function(resp) {
				defer.resolve(resp);
			}).error(function(err) {
				defer.reject(err);
			});
			return defer.promise;
		}
	};
}]);
